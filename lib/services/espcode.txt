float readAnalogVoltage(uint8_t pin) {
  long sum = 0;

  for (int i = 0; i < SAMPLE_COUNT; i++) {
    sum += analogRead(pin);
    delay(SAMPLE_DELAY_MS);
  }

  float avg = (float)sum / SAMPLE_COUNT;
  return avg * (3.3f / 4095.0f);
}

float computeAverage(float buffer[], uint8_t count) {
  if (count == 0) return 0.0;

  float sum = 0.0;
  uint8_t n = min(count, (uint8_t)SMOOTH_WINDOW);

  for (uint8_t i = 0; i < n; i++) {
    sum += buffer[i];
  }

  return sum / n;
}

unsigned long getEpochMillis() {
  time_t now;
  time(&now);

  if (now < 1700000000) return 0;

  return (unsigned long)now * 1000UL;
}

// ============================================================
//  CLASSIFICATION / ZONES
// ============================================================
TurbidityResult classifyTurbidity(float v) {
  TurbidityResult r;

  if (v < turbidityVAirMax) {
    r.quality = 0.0;
    r.status = "AIR (NO WATER DETECTED)";
    r.valid = false;
    return r;
  }

  r.status = (v >= turbidityVClearThreshold) ? "CLEAR WATER" : "DIRTY WATER";
  r.quality = (v - turbidityVDirty) * 100.0f / (turbidityVClear - turbidityVDirty);
  r.quality = constrain(r.quality, 0.0f, 100.0f);
  r.valid = true;

  return r;
}

String getTempZone(float t) {
  if (!tempSensorOK || t < -100.0) return "SENSOR ERROR";
  if (t < tempCriticalLow) return "CRITICAL LOW";
  if (t > tempCriticalHigh) return "CRITICAL HIGH";
  return "OPTIMAL";
}

String getTurbidityZone() {
  if (turbidityStatus == "AIR (NO WATER DETECTED)") return "NO WATER";
  if (!turbiditySensorOK) return "SENSOR ERROR";
  if (smoothedTurbidityQuality < turbidityQualityMin) return "HIGH TURBIDITY";
  if (smoothedTurbidityQuality > turbidityQualityMax) return "CRITICAL HIGH";
  return "OPTIMAL";
}

String getDOZone() {
  if (!ENABLE_DO_SENSOR) return "PLACEHOLDER";
  if (!doSensorOK || dissolvedOxygen < 0.0) return "SENSOR ERROR";
  if (dissolvedOxygen < doCriticalLow) return "CRITICAL LOW";
  if (dissolvedOxygen > doCriticalHigh) return "CRITICAL HIGH";
  return "OPTIMAL";
}

String getPHZone() {
  if (!ENABLE_PH_SENSOR) return "PLACEHOLDER";
  if (!phSensorOK || phLevel < 0.0) return "SENSOR ERROR";

String getWaterLevelZone() {
  if (!ENABLE_WATER_LEVEL_SENSOR) return "PLACEHOLDER";
  if (!waterLevelSensorOK || waterLevelPercent < 0.0) return "SENSOR ERROR";
  if (waterLevelPercent < waterLevelCriticalLow) return "CRITICAL LOW";
  if (waterLevelPercent > waterLevelCriticalHigh) return "CRITICAL HIGH";
  return "OPTIMAL";
}

String getOverallStatus() {
  String tempZone = getTempZone(smoothedTemp);
  String turbidityZone = getTurbidityZone();
  String doZone = getDOZone();
  String phZone = getPHZone();
  String waterLevelZone = getWaterLevelZone();

  if (tempZone == "SENSOR ERROR" || turbidityZone == "SENSOR ERROR") return "SENSOR ERROR";
  if (ENABLE_DO_SENSOR && doZone == "SENSOR ERROR") return "SENSOR ERROR";
  if (ENABLE_PH_SENSOR && phZone == "SENSOR ERROR") return "SENSOR ERROR";
  if (ENABLE_WATER_LEVEL_SENSOR && waterLevelZone == "SENSOR ERROR") return "SENSOR ERROR";

  if (turbidityZone == "NO WATER") return "NO WATER";

  if (tempZone == "CRITICAL LOW" || tempZone == "CRITICAL HIGH") return "WARNING";
  if (turbidityZone == "HIGH TURBIDITY") return "WARNING";
  if (ENABLE_DO_SENSOR && (doZone == "CRITICAL LOW" || doZone == "CRITICAL HIGH")) return "WARNING";
  if (ENABLE_PH_SENSOR && (phZone == "CRITICAL LOW" || phZone == "CRITICAL HIGH")) return "WARNING";
  if (ENABLE_WATER_LEVEL_SENSOR && (waterLevelZone == "CRITICAL LOW" || waterLevelZone == "CRITICAL HIGH")) return "WARNING";

  return "OPTIMAL";
}

void connectFirebase() {
  config.api_key = FIREBASE_API_KEY;
  config.database_url = FIREBASE_DATABASE_URL;
  config.token_status_callback = tokenStatusCallback;

  Firebase.reconnectWiFi(true);

  Serial.print("Signing in to Firebase anonymously... ");

  if (Firebase.signUp(&config, &auth, "", "")) {
    firebaseReady = true;
    Serial.println("OK");
  } else {
    firebaseReady = false;
    Serial.printf("FAILED: %s\n", config.signer.signupError.message.c_str());
  }

  Firebase.begin(&config, &auth);
}

bool readConfigFloatPath(const char* relativePath, float &target, float minValue, float maxValue) {
  String path = String(FIREBASE_CONFIG_PATH) + "/" + relativePath;

  if (!Firebase.RTDB.getFloat(&fbdo, path.c_str())) {
    return false;
  }

  float value = fbdo.floatData();

  if (!isfinite(value) || value < minValue || value > maxValue) {
    Serial.printf("[CONFIG SKIP] %s invalid value: %.3f\n", relativePath, value);
    return false;
  }

  target = value;
  return true;
}

bool readConfigFloat(const char* key, float &target, float minValue, float maxValue) {
  return readConfigFloatPath(key, target, minValue, maxValue);
}

bool readRangeConfig(const char* sensorKey, float &lowTarget, float &highTarget, float minLimit, float maxLimit) {
  float newLow = lowTarget;
  float newHigh = highTarget;

  String minPath = String("ranges/") + sensorKey + "/min";
  String maxPath = String("ranges/") + sensorKey + "/max";

  bool gotMin = readConfigFloatPath(minPath.c_str(), newLow, minLimit, maxLimit);
  bool gotMax = readConfigFloatPath(maxPath.c_str(), newHigh, minLimit, maxLimit);

  if (!gotMin && !gotMax) return false;

  if (newLow >= newHigh) {
    Serial.printf("[CONFIG SKIP] ranges/%s min must be lower than max\n", sensorKey);
    return false;
  }

  lowTarget = newLow;
  highTarget = newHigh;
  return true;
}

  json.set("sensors/ph/unit", "pH");
  json.set("sensors/ph/voltage", phVoltage);
  json.set("sensors/ph/status", phStatus);
  json.set("sensors/ph/zone", getPHZone());
  json.set("sensors/ph/ok", phSensorOK);
  json.set("sensors/ph/placeholder", ENABLE_PH_SENSOR == 0);

  json.set("sensors/waterLevel/value", waterLevelPercent);
  json.set("sensors/waterLevel/unit", "%");
  json.set("sensors/waterLevel/voltage", waterLevelVoltage);
  json.set("sensors/waterLevel/status", waterLevelStatus);
  json.set("sensors/waterLevel/zone", getWaterLevelZone());
  json.set("sensors/waterLevel/ok", waterLevelSensorOK);
  json.set("sensors/waterLevel/placeholder", ENABLE_WATER_LEVEL_SENSOR == 0);

  json.set("overallStatus", getOverallStatus());
  json.set("deviceOnline", true);
  json.set("deviceMillis", (int)millis());
  json.set("timestamp", getEpochMillis());
  json.set("wifiRSSI", WiFi.RSSI());
  json.set("tankName", "Tank A");

  addThresholdsToJson(json);

  return json;
}

void sendLatestToFirebase() {
  if (!firebaseReady || !Firebase.ready()) return;

  FirebaseJson json = buildSensorJson();

  if (Firebase.RTDB.updateNode(&fbdo, FIREBASE_LATEST_PATH, &json)) {
    Serial.println("[FIREBASE] Latest data sent");
  } else {
    Serial.printf("[FIREBASE ERROR] %s\n", fbdo.errorReason().c_str());
  }
}

void sendHistoryToFirebase() {
  if (!firebaseReady || !Firebase.ready()) return;

  FirebaseJson json = buildSensorJson();
  json.set("createdAtMillis", (int)millis());
  json.set("createdAt", getEpochMillis());

  if (Firebase.RTDB.pushJSON(&fbdo, FIREBASE_HISTORY_PATH, &json)) {
    Serial.println("[FIREBASE] History data saved");
  } else {
    Serial.printf("[FIREBASE HISTORY ERROR] %s\n", fbdo.errorReason().c_str());
  }
}

// ============================================================
//  SENSOR PRIMING
// ============================================================
void primeTemperatureBuffer() {
  sensors.requestTemperatures();
  float ft = sensors.getTempCByIndex(0);

  if (ft > MIN_VALID_TEMP && ft < MAX_VALID_TEMP) {
    lastValidTemp = ft;

    for (uint8_t i = 0; i < SMOOTH_WINDOW; i++) {
      tempBuffer[i] = ft;
    }

    tempCount = SMOOTH_WINDOW;
    tempIndex = 0;
    smoothedTemp = ft;
    tempSensorOK = true;
  }
}

void primeTurbidityBuffer() {
  float fv = readAnalogVoltage(TURBIDITY_PIN);
  TurbidityResult tr = classifyTurbidity(fv);

  turbidityVoltage = fv;
  lastValidTurbidityQuality = tr.quality;

  for (uint8_t i = 0; i < SMOOTH_WINDOW; i++) {
    turbidityBuffer[i] = tr.quality;
  }

  turbidityCount = SMOOTH_WINDOW;
  turbidityIndex = 0;
  smoothedTurbidityQuality = tr.quality;
  turbidityStatus = tr.status;
  turbiditySensorOK = tr.valid;
}

// ============================================================
//  SENSOR READ FUNCTIONS
// ============================================================
void readTemperatureSensor() {
  sensors.requestTemperatures();
  float rawTemp = sensors.getTempCByIndex(0);

  if (rawTemp < MIN_VALID_TEMP || rawTemp > MAX_VALID_TEMP) {
    tempSensorOK = false;
    Serial.printf("[TEMP SKIP] out of bounds: %.1f\n", rawTemp);
    return;
  }

  bool accept = true;

  if (lastValidTemp > -100.0) {
    float jump = fabs(rawTemp - lastValidTemp);

    if (jump > TEMP_JUMP_MAX) {
      accept = false;
      Serial.printf("[TEMP SKIP] jump too large: %.2f\n", jump);
    }
  }

  if (accept) {
    tempSkipCount = 0;
    tempBuffer[tempIndex] = rawTemp;
    tempIndex = (tempIndex + 1) % SMOOTH_WINDOW;

    if (tempCount < SMOOTH_WINDOW) tempCount++;

    lastValidTemp = rawTemp;
    tempSensorOK = true;
    smoothedTemp = computeAverage(tempBuffer, tempCount);
  } else {
    tempSkipCount++;

    if (tempSkipCount >= MAX_SKIP_COUNT) {
      Serial.println("[TEMP] Watchdog override — forcing new baseline.");
      lastValidTemp = rawTemp;
      tempSkipCount = 0;
    }
  }
}

void readTurbiditySensor() {
  float voltage = readAnalogVoltage(TURBIDITY_PIN);
  TurbidityResult tr = classifyTurbidity(voltage);

  turbidityVoltage = voltage;

  if (!tr.valid) {
    turbiditySensorOK = false;
    turbidityStatus = tr.status;
    smoothedTurbidityQuality = 0.0;
    Serial.printf("[TURB] %s (V=%.3f)\n", tr.status.c_str(), voltage);
    return;
  }

  bool accept = true;

  if (lastValidTurbidityQuality >= 0.0) {
    float jump = fabs(tr.quality - lastValidTurbidityQuality);

    if (jump > TURBIDITY_QUALITY_JUMP_MAX) {
      accept = false;
      Serial.printf("[TURB SKIP] quality jump too large: %.1f\n", jump);
    }
  }

  if (accept) {
    turbiditySkipCount = 0;
    turbidityBuffer[turbidityIndex] = tr.quality;
    turbidityIndex = (turbidityIndex + 1) % SMOOTH_WINDOW;

    if (turbidityCount < SMOOTH_WINDOW) turbidityCount++;

    lastValidTurbidityQuality = tr.quality;
    turbiditySensorOK = true;
    turbidityStatus = tr.status;
    smoothedTurbidityQuality = computeAverage(turbidityBuffer, turbidityCount);
  } else {
    turbiditySkipCount++;

    if (turbiditySkipCount >= MAX_SKIP_COUNT) {
      Serial.println("[TURB] Watchdog override — forcing new baseline.");
      lastValidTurbidityQuality = tr.quality;
      turbiditySkipCount = 0;
    }
  }
}

void readDissolvedOxygenSensor() {
  if (!ENABLE_DO_SENSOR) {
    dissolvedOxygen = -1.0;
    dissolvedOxygenVoltage = 0.0;
    doSensorOK = false;
    doStatus = "PLACEHOLDER - SENSOR NOT ENABLED";
    return;
  }

  dissolvedOxygenVoltage = readAnalogVoltage(DO_PIN);
  dissolvedOxygen = dissolvedOxygenVoltage * doVoltageScale + doVoltageOffset;
  dissolvedOxygen = constrain(dissolvedOxygen, 0.0f, 30.0f);
  doSensorOK = true;
  doStatus = getDOZone();
}

void readPHSensor() {
  if (!ENABLE_PH_SENSOR) {
    phLevel = -1.0;
    phVoltage = 0.0;
    phSensorOK = false;
    phStatus = "PLACEHOLDER - SENSOR NOT ENABLED";
    return;
  }

  phVoltage = readAnalogVoltage(PH_PIN);
  phLevel = phVoltageSlope * phVoltage + phVoltageIntercept;
  phLevel = constrain(phLevel, 0.0f, 14.0f);
  phSensorOK = true;
  phStatus = getPHZone();
}

void readWaterLevelSensor() {
  if (!ENABLE_WATER_LEVEL_SENSOR) {
    waterLevelPercent = -1.0;
    waterLevelVoltage = 0.0;
    waterLevelSensorOK = false;
    waterLevelStatus = "PLACEHOLDER - SENSOR NOT ENABLED";
    return;
  }

  waterLevelVoltage = readAnalogVoltage(WATER_LEVEL_PIN);
  waterLevelPercent = (waterLevelVoltage - waterLevelVoltageMin) * 100.0f / (waterLevelVoltageMax - waterLevelVoltageMin);
  waterLevelPercent = constrain(waterLevelPercent, 0.0f, 100.0f);
  waterLevelSensorOK = true;
  waterLevelStatus = getWaterLevelZone();
}

void readAllSensors() {
  readTemperatureSensor();
  readTurbiditySensor();
  readDissolvedOxygenSensor();
  readPHSensor();
  readWaterLevelSensor();
}

// ============================================================
//  SETUP
// ============================================================
void setup() {
  Serial.begin(115200);
  delay(500);

  analogReadResolution(12);
  analogSetAttenuation(ADC_11db);

  sensors.begin();

  primeTemperatureBuffer();
  primeTurbidityBuffer();

  connectWiFi();
  initTime();
  connectFirebase();
  syncConfigFromFirebase();

  Serial.println("============================================");
  Serial.println("  CrayCare Monitor + Firebase — Ready");
  Serial.printf("  Latest path : %s\n", FIREBASE_LATEST_PATH);
  Serial.printf("  History path: %s\n", FIREBASE_HISTORY_PATH);
  Serial.printf("  Config path : %s\n", FIREBASE_CONFIG_PATH);
  Serial.println("============================================");
}

// ============================================================
//  LOOP
// ============================================================
void loop() {
  unsigned long now = millis();

  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("[WIFI] Reconnecting...");
    WiFi.disconnect();
    WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
    delay(1000);
    return;
  }

  if (now - lastConfigSyncTime >= CONFIG_SYNC_INTERVAL_MS) {
    lastConfigSyncTime = now;
    syncConfigFromFirebase();
  }

  if (now - lastPollTime >= SENSOR_POLL_MS) {
    lastPollTime = now;

    readAllSensors();

    Serial.printf("[OK] Temp: %.1f C (%s) | Turbidity: %.1f%% (%s) | DO: %.1f (%s) | pH: %.2f (%s) | Level: %.1f%% (%s) | Overall: %s\n",
                  smoothedTemp,
                  getTempZone(smoothedTemp).c_str(),
                  smoothedTurbidityQuality,
                  getTurbidityZone().c_str(),
                  dissolvedOxygen,
                  getDOZone().c_str(),
                  phLevel,
                  getPHZone().c_str(),
                  waterLevelPercent,
                  getWaterLevelZone().c_str(),
                  getOverallStatus().c_str());
  }

  if (now - lastFirebaseSendTime >= FIREBASE_SEND_INTERVAL_MS) {
    lastFirebaseSendTime = now;
    sendLatestToFirebase();
  }

  if (now - lastHistorySendTime >= HISTORY_SEND_INTERVAL_MS) {
    lastHistorySendTime = now;
    sendHistoryToFirebase();
  }
}
